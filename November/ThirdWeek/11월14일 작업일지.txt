def cookie_checking
	@user_cookie = @user.cookie
	@task_cookie = @task.cookie

	if @user_cookie >= @giftcon.price
		@user_cookie.cookie - @giftcon.price
		@task_cookie.cookie + @giftcon.price
		@user.save
		@task.save
	else @user_cookie < @giftcon.price
		false
end

def	cookie_to_task(price)
	cookie - price
	save!
end
